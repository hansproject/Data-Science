from selenium import webdriver 
 
import time 
 
def loginBOT(usr, pas): 
    br = webdriver.Chrome(executable_path=r'C:\\Webdriver\\chromedriver.exe') 
    br.get("https://web.facebook.com/") 
    user = br.find_element_by_css_selector("#email") 
    user.clear() 
    user.send_keys(usr) 
    passwd = br.find_element_by_css_selector("#pass") 
    passwd.clear() 
    passwd.send_keys(pas) 
    btn = br.find_element_by_id('loginbutton') 
    btn.click() 
    time.sleep(15) 
 
if __name__ == "__main__": 
    loginBOT('username', 'password')
